Stata Do-File Templates

cleaning.do — Data Cleaning


********************************************************************************
* DATA CLEANING TEMPLATE
* AUTHOR: Dr Abdulla
* DATE: 2025-10-29
* DESCRIPTION: Handle missings, outliers, duplicates, and validate data.
********************************************************************************

clear all
set more off
set varabbrev off

global project_root "C:/Users/YourName/Dropbox/Projects/YourProject"
global data_raw     "$project_root/01_data/raw"
global data_clean   "$project_root/01_data/clean"
global log          "$project_root/04_log"

capture log close
log using "$log/cleaning_`c(current_date)'.log", replace

*--- Install packages -------------------------------------------------------
foreach pkg in missings winsor2 {
    capture which `pkg'
    if _rc ssc install `pkg', replace
}

*--- Load raw data ----------------------------------------------------------
use "$data_raw/raw_data.dta", clear

*--- Missing values ---------------------------------------------------------
missings report
replace income = . if income == -999
drop if missing(id)

*--- Outliers ---------------------------------------------------------------
winsor2 income age, cuts(1 99) replace
replace income = . if income > 1e6

*--- Duplicates -------------------------------------------------------------
duplicates report id
duplicates drop id, force

*--- Recode & label ---------------------------------------------------------
recode gender (1=0 "Male") (2=1 "Female"), gen(gender_bin)
label var gender_bin "Gender: 0=Male, 1=Female"

*--- Validation -------------------------------------------------------------
assert age >= 0 & age <= 120
codebook, compact

*--- Save -------------------------------------------------------------------
save "$data_clean/cleaned_data.dta", replace

log close

transformation.do — Reshape & Merge


********************************************************************************
* DATA TRANSFORMATION TEMPLATE
* AUTHOR: Dr Abdulla
* DATE: 2025-10-29
********************************************************************************

clear all
set more off

global data_clean     "$project_root/01_data/clean"
global data_transform "$project_root/01_data/transformed"

log using "$log/transformation_`c(current_date)'.log", replace

*--- Load cleaned data ------------------------------------------------------
use "$data_clean/cleaned_data.dta", clear

*--- Generate variables -----------------------------------------------------
gen log_income = log(income + 1)
egen z_age = std(age)
gen income_cat = cond(income < 5e4, 1, cond(income < 1e5, 2, 3))

*--- Reshape: wide to long --------------------------------------------------
reshape long sales_, i(id) j(year)

*--- Merge additional data --------------------------------------------------
merge 1:1 id year using "$data_clean/covariates.dta", nogen
merge m:1 region using "$data_clean/region_info.dta", keepusing(region_name) nogen

*--- Group-wise transformations ---------------------------------------------
bysort region: egen mean_income = mean(income)

*--- Save -------------------------------------------------------------------
save "$data_transform/analysis_ready.dta", replace

log close

measurement.do — Summary & Indices


********************************************************************************
* MEASUREMENT & SUMMARY STATISTICS
* AUTHOR: Dr Abdulla
********************************************************************************

clear all
set more off

global data_transform "$project_root/01_data/transformed"
global output         "$project_root/03_output"

log using "$log/measurement_`c(current_date)'.log", replace

*--- Packages ---------------------------------------------------------------
foreach pkg in estout asdoc {
    capture which `pkg'
    if _rc ssc install `pkg', replace
}

*--- Load data --------------------------------------------------------------
use "$data_transform/analysis_ready.dta", clear

*--- Summary statistics -----------------------------------------------------
summarize income age, detail
tabstat income age, by(gender) stats(mean sd n)

*--- Correlations -----------------------------------------------------------
pwcorr income education assets, sig star(.05)

*--- Composite index --------------------------------------------------------
egen ses_index = rowmean(income education assets)
factor income education assets, pcf
predict ses_pca

*--- Export table -----------------------------------------------------------
estpost tabstat income age education, stats(mean sd min max n) columns(statistics)
esttab using "$output/summary.rtf", cells("mean sd count") replace

log close

mapping.do — Choropleth Maps


********************************************************************************
* GEOSPATIAL MAPPING TEMPLATE
* AUTHOR: Dr Abdulla
* DESCRIPTION: Create publication-ready maps with spmap
********************************************************************************

clear all
set more off

global gis_data "$project_root/01_data/gis"
global output   "$project_root/03_output"

log using "$log/mapping_`c(current_date)'.log", replace

*--- Install packages -------------------------------------------------------
foreach pkg in spmap shp2dta geo2xy {
    capture which `pkg'
    if _rc ssc install `pkg', replace
}

*--- Convert shapefile (run once) -------------------------------------------
shp2dta using "$gis_data/india_adm2.shp", database(india_db) coord(india_coord) genid(id) replace

*--- Merge data -------------------------------------------------------------
use "$data_transform/analysis_ready.dta", clear
merge m:1 district using "india_db.dta", keepusing(id) nogen

*--- Create map -------------------------------------------------------------
spmap income using "india_coord.dta", id(id) ///
    fcolor(Blues) clmethod(quantile) clnumber(5) ///
    title("Income by District") legend(pos(4)) ///
    note("Source: Your Survey + GADM")
graph export "$output/income_map.pdf", replace

log close

Master Do-File & Learning Roadmap

A complete, self-contained do-file combining all examples. Copy-paste into Stata.

Stata_Learning_Complete.do


* ==============================================================================
* STATA LEARNING: COMPLETE DO-FILE (BEGINNER → ADVANCED)
* Combines: Webinars + Free Courses + Real Examples
* Run in Stata 17/18/19 (works in 15+)
* ==============================================================================

clear all
set more off
cd "C:\stata_learning"  // CHANGE TO YOUR FOLDER

* -----------------------------------------------------------------------------
* EXAMPLE 1: READY. SET. GO STATA (Webinar 1)
* -----------------------------------------------------------------------------
sysuse auto, clear
describe
summarize price mpg weight
list make price mpg in 1/5
twoway scatter price mpg, title("Price vs MPG") 

* Save graph
graph export "price_mpg.png", replace

* -----------------------------------------------------------------------------
* EXAMPLE 2: DATA IMPORT & CLEANING (UCLA + Alison Course)
* -----------------------------------------------------------------------------
import excel "https://stats.idre.ucla.edu/stat/data/hsb2.xls", firstrow clear
rename id ID
rename race Race
label variable read "Reading Score"
label define racelbl 1 "Hispanic" 2 "Asian" 3 "African Am" 4 "White"
label values Race racelbl
tab Race

* Recode and generate
gen high_math = (math > 60)
replace high_math = . if missing(math)
tab high_math

* Save cleaned data
save "hsb2_clean.dta", replace

* -----------------------------------------------------------------------------
* EXAMPLE 3: SURVEY DATA (Webinar: Survey Statistics)
* -----------------------------------------------------------------------------
webuse nhanes2f, clear
svyset psuid [pweight=finalwgt], strata(stratid)
svy: mean age
svy: mean bpsystol, over(sex)
svy: regress bpsystol age i.sex

* -----------------------------------------------------------------------------
* EXAMPLE 4: PANEL DATA (Webinar: Panel Data)
* -----------------------------------------------------------------------------
webuse grunfeld, clear
xtset company year
xtdescribe
xtsum invest

* Fixed effects
xtreg invest mvalue kstock, fe
estimates store fe

* Random effects
xtreg invest mvalue kstock, re
estimates store re

* Hausman test
hausman fe re

* -----------------------------------------------------------------------------
* EXAMPLE 5: STRUCTURAL EQUATION MODELING (Webinar: SEM)
* -----------------------------------------------------------------------------
webuse sem_example, clear
sem (Math -> read write) (Science -> math science), standardized
estat eqgof
estat teffects
sem, standardized

* Draw path diagram
estat mindices
* (Use Graph Editor to refine)

* -----------------------------------------------------------------------------
* EXAMPLE 6: GRAPHICS & EXPORT (UCLA + Princeton)
* -----------------------------------------------------------------------------
use hsb2_clean.dta, clear
twoway (scatter read write) (lfit read write), ///
    title("Reading vs Writing Scores") legend(label(1 "Data") label(2 "Fit"))

graph export "read_write_scatter.png", replace

* Multiple graphs
graph combine "price_mpg.png" "read_write_scatter.png", cols(2) title("Combined Graphs")
graph export "combined_graphs.png", replace

* -----------------------------------------------------------------------------
* EXAMPLE 7: AUTOMATION & LOOP (Tips and Tricks Webinar)
* -----------------------------------------------------------------------------
foreach var of varlist read write math science {
    summarize `var'
}
    display "Mean of `var' = " r(mean)
}

* Save summary table
eststo clear
foreach var of varlist read write math science {
    eststo: quietly sum `var'
}
esttab using "summary_table.rtf", cells(mean sd) replace

* -----------------------------------------------------------------------------
* EXAMPLE 8: TIME SERIES (Bonus: LinkedIn Learning)
* -----------------------------------------------------------------------------
webuse air2, clear
tsset t, yearly
tsline air
dfuller air, lags(1)
arima air, ar(1) ma(1)
predict air_fcast, dynamic(t+1)

* -----------------------------------------------------------------------------
* BONUS: REPRODUCIBILITY (Best Practice)
* -----------------------------------------------------------------------------
log using "stata_learning_log.smcl", replace
display "Analysis completed on $S_DATE at $S_TIME"
log close

* ==============================================================================
* END OF DO-FILE
* Practice Tip: Run one section at a time. Modify with your own data!
* ==============================================================================

Example: Spatial Autoregressive Model (SAR) with Map 

// 1. Import shapefile (e.g., US counties)
spshape2dta us_counties.shp, replace

// 2. Load data and merge with shapefile
use us_counties, clear
merge 1:1 _ID using us_counties_shp, nogen

// 3. Declare spatial structure
spset _ID

// 4. Create contiguity weight matrix (queen)
spmat use W using us_counties.spmat, replace
spmat summarize W

// 5. Estimate Spatial Autoregressive Model (SAR)
spregress income education unemployment, gs2sls dvarlag(W)

// 6. Predict spatial lag and residuals
predict yhat, xb
predict resid, residuals

// 7. Visualize results on map
spmap income using us_counties_shp, id(_ID) clmethod(custom) clbreaks(0 20000 40000 60000 80000) ///
    title("Median Income by County") legend(pos(5) size(small))

// Optional: Moran’s I test on residuals
spatgsa resid, weights(W) graph

Requirements: Stata 15+, shapefile (.shp), and data merged on _ID.
