cleaning.do — Advanced Data Cleaning


********************************************************************************
* ADVANCED DATA CLEANING TEMPLATE
* AUTHOR: Dr Abdulla
* DATE: 2025-10-29
* DESCRIPTION: Handle missings, outliers, duplicates, validate, and log errors.
********************************************************************************

clear all
set more off
set varabbrev off
set matsize 11000

global project_root "C:/Users/YourName/Dropbox/Projects/YourProject"
global data_raw     "$project_root/01_data/raw"
global data_clean   "$project_root/01_data/clean"
global log          "$project_root/04_log"

capture log close
log using "$log/cleaning_`c(current_date)'.log", replace text

*--- Install/Require packages -----------------------------------------------
local pkgs "missings winsor2 reghdfe estout"
foreach pkg in `pkgs' {
    capture which `pkg'
    if _rc ssc install `pkg', replace
}

*--- Load raw data ----------------------------------------------------------
use "$data_raw/raw_data.dta", clear

*--- Missing values handling ------------------------------------------------
missings report
missings dropvars, min(50)  // Drop vars with >50% missing
replace income = . if income < 0 | income == 9999  // Invalid codes
drop if missing(id)  // Required ID

*--- Outliers detection & treatment -----------------------------------------
winsor2 income age, cuts(1 99) trim replace
replace income = . if income > 1e7  // Extreme values

*--- Duplicates management --------------------------------------------------
duplicates report id
duplicates tag id, gen(dup)
drop if dup > 0 & _n > 1  // Keep first occurrence
drop dup

*--- Recode, label, & type conversion ---------------------------------------
recode gender (1=0 "Male") (2=1 "Female") (else=.), gen(gender_bin)
label var gender_bin "Binary Gender: 0=Male, 1=Female"
destring str_var, replace force

*--- Data validation & assertions -------------------------------------------
assert age >= 18 & age <= 100 if !missing(age)
isid id  // Check unique ID
codebook, compact problems

*--- Save cleaned dataset ---------------------------------------------------
compress
save "$data_clean/cleaned_data.dta", replace

log close

transformation.do — Data Reshape & Merge


********************************************************************************
* ADVANCED DATA TRANSFORMATION TEMPLATE
* AUTHOR: Dr Abdulla
* DATE: 2025-10-29
********************************************************************************

clear all
set more off

global data_clean     "$project_root/01_data/clean"
global data_transform "$project_root/01_data/transformed"

log using "$log/transformation_`c(current_date)'.log", replace

*--- Load cleaned data ------------------------------------------------------
use "$data_clean/cleaned_data.dta", clear

*--- Generate & transform variables -----------------------------------------
gen log_income = log(income + 1)
egen z_age = std(age)
gen income_cat = autocode(income, 3, 0, .)  // Automatic categories

*--- Reshape data -----------------------------------------------------------
reshape long sales_, i(id) j(year)
reshape wide var_, i(id year) j(category) string

*--- Merge datasets with checks ---------------------------------------------
merge 1:1 id year using "$data_clean/covariates.dta", assert(3) nogen
merge m:1 region using "$data_clean/region_info.dta", keep(1 3) keepusing(region_name) nogen

*--- Group operations & aggregations ----------------------------------------
bysort region: egen mean_income = mean(income)
collapse (mean) avg_* = *, by(region)

*--- Save transformed data --------------------------------------------------
save "$data_transform/analysis_ready.dta", replace

log close

measurement.do — Summary Stats & Indices


********************************************************************************
* ADVANCED MEASUREMENT & SUMMARY STATISTICS
* AUTHOR: Dr Abdulla
********************************************************************************

clear all
set more off

global data_transform "$project_root/01_data/transformed"
global output         "$project_root/03_output"

log using "$log/measurement_`c(current_date)'.log", replace

*--- Packages ---------------------------------------------------------------
local pkgs "estout asdoc tabout"
foreach pkg in `pkgs' {
    capture which `pkg'
    if _rc ssc install `pkg', replace
}

*--- Load data --------------------------------------------------------------
use "$data_transform/analysis_ready.dta", clear

*--- Detailed summary statistics --------------------------------------------
summarize income age, detail sep(0)
tabstat income age, by(gender) stats(mean sd min p25 p50 p75 max n) longstub

*--- Correlations & matrices ------------------------------------------------
pwcorr income education assets, sig star(.05) obs
correlate _all, cov

*--- Indices & factor analysis ----------------------------------------------
alpha income education assets, gen(ses_index) std
factor income education assets, pcf
predict ses_pca, regression

*--- Export to LaTeX/RTF/HTML -----------------------------------------------
estpost tabstat income age education, stats(mean sd min max n) columns(statistics)
esttab using "$output/summary.tex", cells("mean sd count") replace label

log close

mapping.do — Advanced Geospatial Mapping


********************************************************************************
* ADVANCED GEOSPATIAL MAPPING TEMPLATE
* AUTHOR: Dr Abdulla
* DESCRIPTION: Create interactive, publication-ready maps with layers
********************************************************************************

clear all
set more off

global gis_data "$project_root/01_data/gis"
global output   "$project_root/03_output"

log using "$log/mapping_`c(current_date)'.log", replace

*--- Install packages -------------------------------------------------------
local pkgs "spmap shp2dta geo2xy grmap"
foreach pkg in `pkgs' {
    capture which `pkg'
    if _rc ssc install `pkg', replace
}

*--- Convert shapefile (if needed) ------------------------------------------
shp2dta using "$gis_data/india_adm2.shp", database(india_db) coord(india_coord) genid(id) replace gencentroids(c) 

*--- Merge geospatial data --------------------------------------------------
use "$data_transform/analysis_ready.dta", clear
merge m:1 district using "india_db.dta", keepusing(id _CX _CY) nogen

*--- Advanced map with layers & legends -------------------------------------
spmap income using "india_coord.dta", id(id) ///
    fcolor(Blues) clmethod(quantile) clnumber(5) ///
    polygon(data("india_coord.dta") ocolor(red) osize(thin)) ///
    point(x(_CX) y(_CY) proportional(pop) fcolor(Greens)) ///
    title("Income Distribution by District") subtitle("2025 Analysis") ///
    legend(pos(7) size(medium) title("Quintiles")) note("Source: Survey + GADM")
graph export "$output/advanced_map.pdf", replace

log close

Stata_Learning_Complete.do — Master Combined Examples


* ==============================================================================
* STATA LEARNING: COMPLETE DO-FILE (BEGINNER → ADVANCED)
* Combines: Webinars + Free Courses + Real Examples
* Run in Stata 17/18/19 (works in 15+)
* ==============================================================================

clear all
set more off
cd "C:\stata_learning"  // CHANGE TO YOUR FOLDER

* -----------------------------------------------------------------------------
* EXAMPLE 1: READY. SET. GO STATA (Webinar 1)
* -----------------------------------------------------------------------------
sysuse auto, clear
describe
summarize price mpg weight
list make price mpg in 1/5
twoway scatter price mpg, title("Price vs MPG") 

* Save graph
graph export "price_mpg.png", replace

* -----------------------------------------------------------------------------
* EXAMPLE 2: DATA IMPORT & CLEANING (UCLA + Alison Course)
* -----------------------------------------------------------------------------
import excel "https://stats.idre.ucla.edu/stat/data/hsb2.xls", firstrow clear
rename id ID
label variable read "Reading Score"
label define racelbl 1 "Hispanic" 2 "Asian" 3 "African Am" 4 "White"
label values Race racelbl
tab Race

* Recode and generate
gen high_math = (math > 60)
replace high_math = . if missing(math)
tab high_math

* Save cleaned data
save "hsb2_clean.dta", replace

* -----------------------------------------------------------------------------
* EXAMPLE 3: SURVEY DATA (Webinar: Survey Statistics)
* -----------------------------------------------------------------------------
webuse nhanes2f, clear
svyset psuid [pweight=finalwgt], strata(stratid)
svy: mean age
svy: mean bpsystol, over(sex)
svy: regress bpsystol age i.sex

* -----------------------------------------------------------------------------
* EXAMPLE 4: PANEL DATA (Webinar: Panel Data)
* -----------------------------------------------------------------------------
webuse grunfeld, clear
xtset company year
xtdescribe
xtsum invest

* Fixed effects
xtreg invest mvalue kstock, fe
estimates store fe

* Random effects
xtreg invest mvalue kstock, re
estimates store re

* Hausman test
hausman fe re

* -----------------------------------------------------------------------------
* EXAMPLE 5: STRUCTURAL EQUATION MODELING (Webinar: SEM)
* -----------------------------------------------------------------------------
webuse sem_example, clear
sem (Math -> read write) (Science -> math science), standardized
estat eqgof
estat teffects
sem, standardized

* Draw path diagram
estat mindices
* (Use Graph Editor to refine)

* -----------------------------------------------------------------------------
* EXAMPLE 6: GRAPHICS & EXPORT (UCLA + Princeton)
* -----------------------------------------------------------------------------
use hsb2_clean.dta, clear
twoway (scatter read write) (lfit read write), ///
    title("Reading vs Writing Scores") legend(label(1 "Data") label(2 "Fit"))

graph export "read_write_scatter.png", replace

* Multiple graphs
graph combine "price_mpg.png" "read_write_scatter.png", cols(2) title("Combined Graphs")
graph export "combined_graphs.png", replace

* -----------------------------------------------------------------------------
* EXAMPLE 7: AUTOMATION & LOOP (Tips and Tricks Webinar)
* -----------------------------------------------------------------------------
foreach var of varlist read write math science {
    summarize `var'
    display "Mean of `var' = " r(mean)
}

* Save summary table
eststo clear
foreach var of varlist read write math science {
    eststo: quietly sum `var'
}
esttab using "summary_table.rtf", cells(mean sd) replace

* -----------------------------------------------------------------------------
* EXAMPLE 8: TIME SERIES (Bonus: LinkedIn Learning)
* -----------------------------------------------------------------------------
webuse air2, clear
tsset t, yearly
tsline air
dfuller air, lags(1)
arima air, ar(1) ma(1)
predict air_fcast, dynamic(t+1)

* -----------------------------------------------------------------------------
* BONUS: REPRODUCIBILITY (Best Practice)
* -----------------------------------------------------------------------------
log using "stata_learning_log.smcl", replace
display "Analysis completed on $S_DATE at $S_TIME"
log close

* ==============================================================================
* END OF DO-FILE
* Practice Tip: Run one section at a time. Modify with your own data!
* ==============================================================================

clean_transform.ipynb — Cleaning + Transformation

# Jupyter Notebook Example
# Install: pip install stata_kernel ipystata

%%stata
use raw_data.dta, clear
missings report
winsor2 income, cuts(1 99) trim
gen log_inc = log(income + 1)
reshape long var_, i(id) j(time)
save cleaned_transformed.dta, replace

# Python integration
import pandas as pd
df = pd.read_stata('cleaned_transformed.dta')
df.plot(x='time', y='log_inc')
