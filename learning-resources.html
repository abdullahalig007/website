cleaning.do — Data Cleaning


********************************************************************************
* DATA CLEANING TEMPLATE
* AUTHOR: Dr Abdulla
* DATE: 2025-10-29
* DESCRIPTION: Handle missings, outliers, duplicates, and validate data.
********************************************************************************

clear all
set more off
set varabbrev off

global project_root "C:/Users/YourName/Dropbox/Projects/YourProject"
global data_raw     "$project_root/01_data/raw"
global data_clean   "$project_root/01_data/clean"
global log          "$project_root/04_log"

capture log close
log using "$log/cleaning_`c(current_date)'.log", replace

*--- Install packages -------------------------------------------------------
foreach pkg in missings winsor2 {
    capture which `pkg'
    if _rc ssc install `pkg', replace
}

*--- Load raw data ----------------------------------------------------------
use "$data_raw/raw_data.dta", clear

*--- Missing values ---------------------------------------------------------
missings report
replace income = . if income == -999
drop if missing(id)

*--- Outliers ---------------------------------------------------------------
winsor2 income age, cuts(1 99) replace
replace income = . if income > 1e6

*--- Duplicates -------------------------------------------------------------
duplicates report id
duplicates drop id, force

*--- Recode & label ---------------------------------------------------------
recode gender (1=0 "Male") (2=1 "Female"), gen(gender_bin)
label var gender_bin "Gender: 0=Male, 1=Female"

*--- Validation -------------------------------------------------------------
assert age >= 0 & age <= 120
codebook, compact

*--- Save -------------------------------------------------------------------
save "$data_clean/cleaned_data.dta", replace

log close

transformation.do — Reshape & Merge


********************************************************************************
* DATA TRANSFORMATION TEMPLATE
* AUTHOR: Dr Abdulla
* DATE: 2025-10-29
********************************************************************************

clear all
set more off

global data_clean     "$project_root/01_data/clean"
global data_transform "$project_root/01_data/transformed"

log using "$log/transformation_`c(current_date)'.log", replace

*--- Load cleaned data ------------------------------------------------------
use "$data_clean/cleaned_data.dta", clear

*--- Generate variables -----------------------------------------------------
gen log_income = log(income + 1)
egen z_age = std(age)
gen income_cat = cond(income < 5e4, 1, cond(income < 1e5, 2, 3))

*--- Reshape: wide to long --------------------------------------------------
reshape long sales_, i(id) j(year)

*--- Merge additional data --------------------------------------------------
merge 1:1 id year using "$data_clean/covariates.dta", nogen
merge m:1 region using "$data_clean/region_info.dta", keepusing(region_name) nogen

*--- Group-wise transformations ---------------------------------------------
bysort region: egen mean_income = mean(income)

*--- Save -------------------------------------------------------------------
save "$data_transform/analysis_ready.dta", replace

log close

measurement.do — Summary & Indices


********************************************************************************
* MEASUREMENT & SUMMARY STATISTICS
* AUTHOR: Dr Abdulla
********************************************************************************

clear all
set more off

global data_transform "$project_root/01_data/transformed"
global output         "$project_root/03_output"

log using "$log/measurement_`c(current_date)'.log", replace

*--- Packages ---------------------------------------------------------------
foreach pkg in estout asdoc {
    capture which `pkg'
    if _rc ssc install `pkg', replace
}

*--- Load data --------------------------------------------------------------
use "$data_transform/analysis_ready.dta", clear

*--- Summary statistics -----------------------------------------------------
summarize income age, detail
tabstat income age, by(gender) stats(mean sd n)

*--- Correlations -----------------------------------------------------------
pwcorr income education assets, sig star(.05)

*--- Composite index --------------------------------------------------------
egen ses_index = rowmean(income education assets)
factor income education assets, pcf
predict ses_pca

*--- Export table -----------------------------------------------------------
estpost tabstat income age education, stats(mean sd min max n) columns(statistics)
esttab using "$output/summary.rtf", cells("mean sd count") replace

log close

mapping.do — Choropleth Maps


********************************************************************************
* GEOSPATIAL MAPPING TEMPLATE
* AUTHOR: Dr Abdulla
* DESCRIPTION: Create publication-ready maps with spmap
********************************************************************************

clear all
set more off

global gis_data "$project_root/01_data/gis"
global output   "$project_root/03_output"

log using "$log/mapping_`c(current_date)'.log", replace

*--- Install packages -------------------------------------------------------
foreach pkg in spmap shp2dta geo2xy {
    capture which `pkg'
    if _rc ssc install `pkg', replace
}

*--- Convert shapefile (run once) -------------------------------------------
shp2dta using "$gis_data/india_adm2.shp", database(india_db) coord(india_coord) genid(id) replace

*--- Merge data -------------------------------------------------------------
use "$data_transform/analysis_ready.dta", clear
merge m:1 district using "india_db.dta", keepusing(id) nogen

*--- Create map -------------------------------------------------------------
spmap income using "india_coord.dta", id(id) ///
    fcolor(Blues) clmethod(quantile) clnumber(5) ///
    title("Income by District") legend(pos(4)) ///
    note("Source: Your Survey + GADM")
graph export "$output/income_map.pdf", replace

log close
Download mapping.do
