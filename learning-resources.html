cleaning.do — Advanced Data Cleaning


********************************************************************************
* ADVANCED DATA CLEANING TEMPLATE
* AUTHOR: Dr Abdulla
* DATE: 2025-10-29
* DESCRIPTION: Handle missings, outliers, duplicates, validate, and log errors.
********************************************************************************

clear all
set more off
set varabbrev off
set matsize 11000

global project_root "C:/Users/YourName/Dropbox/Projects/YourProject"
global data_raw     "$project_root/01_data/raw"
global data_clean   "$project_root/01_data/clean"
global log          "$project_root/04_log"

capture log close
log using "$log/cleaning_`c(current_date)'.log", replace text

*--- Install/Require packages -----------------------------------------------
local pkgs "missings winsor2 reghdfe estout"
foreach pkg in `pkgs' {
    capture which `pkg'
    if _rc ssc install `pkg', replace
}

*--- Load raw data ----------------------------------------------------------
use "$data_raw/raw_data.dta", clear

*--- Missing values handling ------------------------------------------------
missings report
missings dropvars, min(50)  // Drop vars with >50% missing
replace income = . if income < 0 | income == 9999  // Invalid codes
drop if missing(id)  // Required ID

*--- Outliers detection & treatment -----------------------------------------
winsor2 income age, cuts(1 99) trim replace
replace income = . if income > 1e7  // Extreme values

*--- Duplicates management --------------------------------------------------
duplicates report id
duplicates tag id, gen(dup)
drop if dup > 0 & _n > 1  // Keep first occurrence
drop dup

*--- Recode, label, & type conversion ---------------------------------------
recode gender (1=0 "Male") (2=1 "Female") (else=.), gen(gender_bin)
label var gender_bin "Binary Gender: 0=Male, 1=Female"
destring str_var, replace force

*--- Data validation & assertions -------------------------------------------
assert age >= 18 & age <= 100 if !missing(age)
isid id  // Check unique ID
codebook, compact problems

*--- Save cleaned dataset ---------------------------------------------------
compress
save "$data_clean/cleaned_data.dta", replace

log close

Stata_Learning_Complete.do — Master Combined Examples


* ==============================================================================
* STATA LEARNING: COMPLETE DO-FILE (BEGINNER → ADVANCED)
* Combines: Webinars + Free Courses + Real Examples
* Run in Stata 17/18/19 (works in 15+)
* ==============================================================================

clear all
set more off
cd "C:\stata_learning"  // CHANGE TO YOUR FOLDER

* EXAMPLE 1: READY. SET. GO STATA
sysuse auto, clear
describe
summarize price mpg weight
list make price mpg in 1/5
twoway scatter price mpg, title("Price vs MPG") 
graph export "price_mpg.png", replace

* EXAMPLE 2: DATA IMPORT & CLEANING
import excel "https://stats.idre.ucla.edu/stat/data/hsb2.xls", firstrow clear
rename id ID
label variable read "Reading Score"
label define racelbl 1 "Hispanic" 2 "Asian" 3 "African Am" 4 "White"
label values Race racelbl
tab Race
gen high_math = (math > 60)
replace high_math = . if missing(math)
tab high_math
save "hsb2_clean.dta", replace

* EXAMPLE 3: SURVEY DATA
webuse nhanes2f, clear
svyset psuid [pweight=finalwgt], strata(stratid)
svy: mean age
svy: mean bpsystol, over(sex)
svy: regress bpsystol age i.sex

* EXAMPLE 4: PANEL DATA
webuse grunfeld, clear
xtset company year
xtdescribe
xtsum invest
xtreg invest mvalue kstock, fe
estimates store fe
xtreg invest mvalue kstock, re
estimates store re
hausman fe re

* EXAMPLE 5: SEM
webuse sem_example, clear
sem (Math -> read write) (Science -> math science), standardized
estat eqgof
estat teffects

* EXAMPLE 6: GRAPHICS
use hsb2_clean.dta, clear
twoway (scatter read write) (lfit read write), title("Reading vs Writing Scores")
graph export "read_write_scatter.png", replace
graph combine "price_mpg.png" "read_write_scatter.png", cols(2)
graph export "combined_graphs.png", replace

* EXAMPLE 7: AUTOMATION
foreach var of varlist read write math science {
    summarize `var'
    display "Mean of `var' = " r(mean)
}
eststo clear
foreach var of varlist read write math science {
    eststo: quietly sum `var'
}
esttab using "summary_table.rtf", cells(mean sd) replace

* EXAMPLE 8: TIME SERIES
webuse air2, clear
tsset t, yearly
tsline air
dfuller air, lags(1)
arima air, ar(1) ma(1)
predict air_fcast, dynamic(t+1)

log using "stata_learning_log.smcl", replace
display "Analysis completed on $S_DATE at $S_TIME"
log close
